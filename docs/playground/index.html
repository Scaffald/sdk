Job Operations​

  • List Jobs
  • Get Single Job
  • Search Jobs
  • Create Job
// List all jobs with filters
const jobs = await client.jobs.list({
  page: 1,
  per_page: 20,
  filters: {
    is_remote: true,
    employment_type: 'full_time',
  },
});

console.log(`Found ${jobs.meta.total} jobs`);
jobs.data.forEach(job => {
  console.log(`${job.title} at ${job.company_name}`);
});
// Get a specific job by ID
const job = await client.jobs.get({
  id: 'job_123',
});

console.log(job.title);
console.log(job.description);
console.log(job.requirements);
// Search for jobs by keyword
const results = await client.jobs.search({
  query: 'Software Engineer',
  location: 'San Francisco',
  page: 1,
  per_page: 10,
});

console.log(`Found ${results.data.length} matching jobs`);
// Create a new job posting
const newJob = await client.jobs.create({
  title: 'Senior Software Engineer',
  company_name: 'Tech Co',
  description: 'We are hiring...',
  requirements: ['5+ years experience', 'React expertise'],
  employment_type: 'full_time',
  is_remote: true,
  salary_min: 150000,
  salary_max: 200000,
});

console.log(`Created job: ${newJob.id}`);

React Integration​

  • Provider Setup
  • Using Hooks
  • Mutations
import { ScaffaldProvider } from '@scaffald/sdk/react';
import { QueryClient, QueryClientProvider } from '@tanstack/react-query';

const queryClient = new QueryClient();

function App() {
  return (
    <QueryClientProvider client={queryClient}>
      <ScaffaldProvider
        config={{
          apiKey: import.meta.env.VITE_SCAFFALD_API_KEY,
        }}
      >
        <YourApp />
      </ScaffaldProvider>
    </QueryClientProvider>
  );
}
import { useJobs, useJob } from '@scaffald/sdk/react';

function JobsList() {
  const { data, isLoading, error } = useJobs({
    page: 1,
    per_page: 20,
  });

  if (isLoading) return <div>Loading...</div>;
  if (error) return <div>Error: {error.message}</div>;

  return (
    <div>
      {data?.data.map(job => (
        <JobCard key={job.id} job={job} />
      ))}
    </div>
  );
}
import { useCreateApplication } from '@scaffald/sdk/react';
import { useQueryClient } from '@tanstack/react-query';

function ApplyButton({ jobId }: { jobId: string }) {
  const queryClient = useQueryClient();
  const createApplication = useCreateApplication({
    onSuccess: () => {
      // Invalidate and refetch
      queryClient.invalidateQueries(['applications']);
      alert('Application submitted!');
    },
  });

  const handleApply = () => {
    createApplication.mutate({
      job_id: jobId,
      profile_id: 'profile_123',
      cover_letter: 'I am very interested...',
    });
  };

  return (
    <button 
      onClick={handleApply}
      disabled={createApplication.isPending}
    >
      {createApplication.isPending ? 'Submitting...' : 'Apply Now'}
    </button>
  );
}

OAuth Authentication​

  • 1. Authorization
  • 2. Handle Callback
  • 3. Use Token
  • 4. Refresh Token
import Scaffald from '@scaffald/sdk';

const client = new Scaffald({
  clientId: 'your_client_id',
  clientSecret: 'your_client_secret',
  redirectUri: 'https://yourapp.com/callback',
});

// Generate authorization URL
const authUrl = client.oauth.getAuthorizationUrl({
  scope: ['read:jobs', 'write:applications'],
  state: crypto.randomUUID(),
});

// Redirect user to authorization
window.location.href = authUrl;
// In your callback route handler
const urlParams = new URLSearchParams(window.location.search);
const code = urlParams.get('code');
const state = urlParams.get('state');

// Exchange code for tokens
const tokens = await client.oauth.exchangeCode({
  code,
  state,
});

// Store tokens securely
sessionStorage.setItem('access_token', tokens.access_token);
sessionStorage.setItem('refresh_token', tokens.refresh_token);
// Initialize client with access token
const client = new Scaffald({
  accessToken: sessionStorage.getItem('access_token'),
});

// Make authenticated requests
const profile = await client.profiles.me();
console.log(`Logged in as: ${profile.first_name}`);
// When access token expires, refresh it
async function refreshAccessToken() {
  const refreshToken = sessionStorage.getItem('refresh_token');
  
  const tokens = await client.oauth.refreshToken({
    refresh_token: refreshToken,
  });
  
  // Update stored tokens
  sessionStorage.setItem('access_token', tokens.access_token);
  sessionStorage.setItem('refresh_token', tokens.refresh_token);
  
  return tokens.access_token;
}

Error Handling​

  • Basic Error Handling
  • Validation Errors
  • Retry Logic
import { ScaffaldError } from '@scaffald/sdk';

try {
  const job = await client.jobs.get({ id: 'invalid_id' });
} catch (error) {
  if (error instanceof ScaffaldError) {
    console.error(`API Error: ${error.message}`);
    console.error(`Status: ${error.statusCode}`);
  } else {
    console.error('Unexpected error:', error);
  }
}
import { ValidationError } from '@scaffald/sdk';

try {
  await client.applications.create({
    job_id: 'job_123',
    // Missing required profile_id
  });
} catch (error) {
  if (error instanceof ValidationError) {
    console.error('Validation failed:');
    error.errors.forEach(err => {
      console.error(`- ${err.field}: ${err.message}`);
    });
  }
}
async function fetchWithRetry<T>(
  fn: () => Promise<T>,
  maxRetries = 3
): Promise<T> {
  for (let i = 0; i < maxRetries; i++) {
    try {
      return await fn();
    } catch (error) {
      const isLastAttempt = i === maxRetries - 1;
      const isServerError = error.statusCode >= 500;
      
      if (isLastAttempt || !isServerError) {
        throw error;
      }
      
      // Exponential backoff
      await new Promise(resolve => 
        setTimeout(resolve, Math.pow(2, i) * 1000)
      );
    }
  }
  throw new Error('Max retries exceeded');
}

// Usage
const jobs = await fetchWithRetry(() => client.jobs.list());

Pagination​

// Automatic pagination through all pages
async function getAllJobs() {
  const allJobs = [];
  let page = 1;
  let hasMore = true;

  while (hasMore) {
    const response = await client.jobs.list({ page, per_page: 50 });
    allJobs.push(...response.data);
    hasMore = response.meta.has_next;
    page++;
  }

  return allJobs;
}

const allJobs = await getAllJobs();
console.log(`Total jobs: ${allJobs.length}`);

Webhook Verification​

import { verifyWebhookSignature } from '@scaffald/sdk';

// In your webhook handler
app.post('/webhooks/scaffald', (req, res) => {
  const signature = req.headers['x-scaffald-signature'];
  const payload = JSON.stringify(req.body);
  const secret = process.env.WEBHOOK_SECRET;

  const isValid = verifyWebhookSignature(payload, signature, secret);

  if (!isValid) {
    return res.status(401).json({ error: 'Invalid signature' });
  }

  // Process webhook
  const event = req.body;
  console.log(`Received ${event.type} event`);
  
  res.json({ received: true });
});
